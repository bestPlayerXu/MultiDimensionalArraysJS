# Multi Dimensional Arrays

```
const MultiDimensionalArrays = require('MultiDimensionalArrayJS');
var 3dArray = new MultiDimensionalArrays([2, 5, 4]);  //x length: 2; y length: 5; z length: 4
3dArray.set('value', [1, 0, 2]);                      //set(value, point)
console.log(3dArray.get([1, 0, 2]));                  //get(point)
const MultiDimensionalArrays = require('MultiDimensionalArrayJS');
var 3dArray = new MultiDimensionalArrays([2, 5, 4]);  //x length: 2; y length: 5; z length: 4
```
